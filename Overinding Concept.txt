        p.show();
        System.out.println(p.a);
    }
}

=====>if we want to access the child class varibles then we use the [Downcasting]


class HelloWorld {

    public static void main(String[] args) {
         
        parent p=new child(); //upcasting
        p.show();
        System.out.println("parentclass varible"+"->"+p.a);
        
        child c = (child) p;
        System.out.println("child class varible"+"->"+c.a);//down casting
    }
}

class parent
{
    int a=10;
     void show()
    {
        System.out.println("parent class methode");
    }
}
class child extends parent{
    int a=20;
     void show()
    {
        System.out.println("child  class methode");
    }
}

	parentclass varible->10
